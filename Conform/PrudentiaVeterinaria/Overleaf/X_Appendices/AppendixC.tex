\chapter{Glossary}
\label{app:Glossary}

\begin{itemize}
  \item \textbf{RCVS}: Royal College of Veterinary Surgeons, the statutory regulator for veterinary surgeons and veterinary nurses in the UK.
  \item \textbf{PSS}: Practice Standards Scheme, the RCVS accreditation and quality framework for veterinary practices.
  \item \textbf{VMD}: Veterinary Medicines Directorate, UK regulator for veterinary medicines.
  \item \textbf{VMR}: Veterinary Medicines Regulations, governing authorisation, supply, prescribing, and use of veterinary medicines.
  \item \textbf{RPS}: Radiation Protection Supervisor, local role for compliance with radiation local rules.
  \item \textbf{RPA}: Radiation Protection Adviser, appointed specialist adviser under radiation legislation.
  \item \textbf{IRR17}: Ionising Radiations Regulations 2017.
  \item \textbf{IRMER}: Ionising Radiation (Medical Exposure) Regulations; relevant principles may apply in veterinary contexts.
  \item \textbf{COSHH}: Control of Substances Hazardous to Health Regulations 2002.
  \item \textbf{RIDDOR}: Reporting of Injuries, Diseases and Dangerous Occurrences Regulations 2013.
  \item \textbf{MDR}: Misuse of Drugs Regulations 2001.
  \item \textbf{AWA}: Animal Welfare Act 2006 and devolved equivalents.
  \item \textbf{ABPR}: Animal By-Products Regulations framework.
  \item \textbf{ASPA}: Animals (Scientific Procedures) Act 1986.
  \item \textbf{VDS}: Veterinary Defence Society.
  \item \textbf{SEA}: Significant Event Audit.
  \item \textbf{EMS}: Extra-Mural Studies placements for veterinary students.
  \item \textbf{BSAVA}: British Small Animal Veterinary Association.
  \item \textbf{AVA}: Association of Veterinary Anaesthetists.
  \item \textbf{OOH}: Out-of-hours clinical service provision.
  \item \textbf{BOAS}: Brachycephalic Obstructive Airway Syndrome.
  \item \textbf{APHA}: Animal and Plant Health Agency.
  \item \textbf{DEFRA}: Department for Environment, Food and Rural Affairs.
\end{itemize}
