\chapter*{Appendix J: The Three Lines Model — Reference Card}
\label{chap:Appendix J: The Three Lines Model — Reference Card}

\section*{First Line — Management}
\paragraph{\textit{Who:} Business units, operational management, front-line staff}
\paragraph{\textit{Role in risk assessment:}}
\begin{itemize}
  \item{Primary ownership of risks within their area.}
  \item{Conducting day-to-day risk assessments.}
  \item{Implementing and operating controls.}
  \item{Monitoring risk indicators}
  \item{Escalating concerns through governance channels.}
  \item{Maintaining accurate risk register entries.}
\end{itemize}
\paragraph{\textit{Accountability:} Risk owners are accountable for the accuracy and currency of their risk assessments and the effectiveness of their controls}

\section*{Second Line — Risk and Compliance Functions}
\paragraph{\textit{Who:} Risk management function, compliance function, legal function, information security function, financial control}
\paragraph{\textit{Role in risk assessment:}}
\begin{itemize}
  \item{Designing and maintaining the risk assessment framework and methodology.}
  \item{Setting risk appetite in conjunction with the board.}
  \item{Providing independent challenge to first-line risk assessments.}
  \item{Aggregating risk information across the organisation.}
  \item{Reporting to governance bodies.}
  \item{Conducting specialist risk assessments.}
  \item{Monitoring the effectiveness of the overall risk management programme.}
\end{itemize}
\paragraph{\textit{Accountability:} Second-line functions are accountable for the quality of the risk framework and the rigour of their oversight and challenge}

\section*{Third Line — Internal Audit}
\paragraph{\textit{Who:} Internal audit function}
\paragraph{\textit{Role in risk assessment:}}
\begin{itemize}
  \item{Providing independent, objective assurance on the adequacy and effectiveness of the risk management framework.}
  \item{Assessing whether the first and second lines are functioning as intended.}
  \item{Identifying gaps and weaknesses in the risk framework.}
  \item{Reporting findings to the audit committee and board.}
  \item{Following up on agreed management actions}
\end{itemize}
\paragraph{\textit{Accountability:} Internal audit is accountable to the audit committee and board for the quality and independence of its assurance activity}
